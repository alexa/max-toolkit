345 inline bool FocusManagerInterface::Activity::notifyObserver(FocusState focus, MixingBehavior behavior) {
346  if (m_channelObserver) {
347  MixingBehavior overrideBehavior{behavior};
348  // If the activity/channelObserver is already paused , do not duck
349  if ((MixingBehavior::MUST_PAUSE == getMixingBehavior()) && (MixingBehavior::MAY_DUCK == behavior)) {
350  overrideBehavior = MixingBehavior::MUST_PAUSE;
351  }
352 
353  m_channelObserver->onFocusChanged(focus, overrideBehavior);
354  // Set the current mixing behavior that the observer received.
355  setMixingBehavior(overrideBehavior);
356  return true;
357  }
358  return false;
359 }
