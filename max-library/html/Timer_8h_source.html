237 template <typename Rep, typename Period, typename Task, typename... Args>
238 bool Timer::start(
239  const std::chrono::duration<Rep, Period>& delay,
240  const std::chrono::duration<Rep, Period>& period,
241  PeriodType periodType,
242  size_t maxCount,
243  Task task,
244  Args&&... args) {
245  if (delay < std::chrono::duration<Rep, Period>::zero()) {
246  // logger::acsdkError(logger::LogEntry(Timer::getTag(), "startFailed").d("reason", "negativeDelay"));
247  return false;
248  }
249  if (period < std::chrono::duration<Rep, Period>::zero()) {
250  // logger::acsdkError(logger::LogEntry(Timer::getTag(), "startFailed").d("reason", "negativePeriod"));
251  return false;
252  }
253 
254  // Can't start if already running.
255  if (!activate()) {
256  // logger::acsdkError(logger::LogEntry(Timer::getTag(), "startFailed").d("reason",
257  // "timerAlreadyActive"));
258  return false;
259  }
260 
261  // Lock for m_thread.join and m_thread assignment
262  std::lock_guard<std::mutex> joinLock(m_joinMutex);
263 
264  // Join old timer thread (if any).
265  if (m_thread.joinable()) {
266  m_thread.join();
267  }
268 
269  // Remove arguments from the task's type by binding the arguments to the task.
270  using BoundTaskType = decltype(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
271  auto boundTask = std::make_shared<BoundTaskType>(std::bind(std::forward<Task>(task), std::forward<Args>(args)...));
272 
273  // Remove the return type from the task by wrapping it in a lambda with no return value.
274  auto translatedTask = [boundTask]() { boundTask->operator()(); };
275 
276  // Kick off the new timer thread.
277  m_thread = std::thread{
278  std::bind(&Timer::callTask<Rep, Period>, this, delay, period, periodType, maxCount, translatedTask)};
279 
280  return true;
281 }

293 template <typename Rep, typename Period, typename Task, typename... Args>
294 auto Timer::start(const std::chrono::duration<Rep, Period>& delay, Task task, Args&&... args)
295  -> std::future<decltype(task(args...))> {
296  // Can't start if already running.
297  if (!activate()) {
298  // logger::acsdkError(logger::LogEntry(Timer::getTag(), "startFailed").d("reason",
299  // "timerAlreadyActive"));
300  using FutureType = decltype(task(args...));
301  return std::future<FutureType>();
302  }
303 
304  // Lock for m_thread.join and m_thread assignment
305  std::lock_guard<std::mutex> joinLock(m_joinMutex);
306 
307  // Join old timer thread (if any).
308  if (m_thread.joinable()) {
309  m_thread.join();
310  }
311 
312  // Remove arguments from the task's type by binding the arguments to the task.
313  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
314 
315  /*
316  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
317  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
318  * will then return a future of the correct type.
319  */
320  using PackagedTaskType = std::packaged_task<decltype(boundTask())()>;
321  auto packagedTask = std::make_shared<PackagedTaskType>(boundTask);
322 
323  // Remove the return type from the task by wrapping it in a lambda with no return value.
324  auto translatedTask = [packagedTask]() { packagedTask->operator()(); };
325 
326  // Kick off the new timer thread.
327  static const size_t once = 1;
328  m_thread = std::thread{
329  std::bind(&Timer::callTask<Rep, Period>, this, delay, delay, PeriodType::ABSOLUTE, once, translatedTask)};
330 
331  return packagedTask->get_future();
332 }

334 template <typename Rep, typename Period>
335 void Timer::callTask(
336  std::chrono::duration<Rep, Period> delay,
337  std::chrono::duration<Rep, Period> period,
338  PeriodType periodType,
339  size_t maxCount,
340  std::function<void()> task) {
341  // Use FinallyGuard to ensure that m_stopping and m_running get reset on completion or exception
342  utils::error::FinallyGuard guard{[this] {
343  m_stopping = false;
344  m_running = false;
345  return;
346  }};
347 
348  // Timepoint to measure delay/period against.
349  auto now = std::chrono::steady_clock::now();
350 
351  // Flag indicating whether we've drifted off schedule.
352  bool offSchedule = false;
353 
354  for (size_t count = 0; maxCount == FOREVER || count < maxCount; ++count) {
355  auto waitTime = (0 == count) ? delay : period;
356  {
357  std::unique_lock<std::mutex> lock(m_waitMutex);
358 
359  // Wait for stop() or a delay/period to elapse.
360  if (m_waitCondition.wait_until(lock, now + waitTime, [this]() { return m_stopping; })) {
361  return;
362  }
363  }
364 
365  switch (periodType) {
366  case PeriodType::ABSOLUTE:
367  // Update our estimate of where we should be after the delay.
368  now += waitTime;
369 
370  // Run the task if we're still on schedule.
371  if (!offSchedule) {
372  task();
373  }
374 
375  // If the task runtime put us off schedule, skip the next task run.
376  if (now + period < std::chrono::steady_clock::now()) {
377  offSchedule = true;
378  } else {
379  offSchedule = false;
380  }
381  break;
382 
383  case PeriodType::RELATIVE:
384  task();
385  now = std::chrono::steady_clock::now();
386  break;
387  }
388  }
389 }
