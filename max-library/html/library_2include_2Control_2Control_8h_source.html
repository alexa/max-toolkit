66  template <typename Task, typename... Args>
67  static std::shared_ptr<ControlTask> createTask(Task task, Args&&... args) {
68  // Remove arguments from the tasks type by binding the arguments to the task.
69  auto boundTask = std::bind(std::forward<Task>(task), std::forward<Args>(args)...);
70 
71  /*
72  * Create a std::packaged_task with the correct return type. The decltype only returns the return value of the
73  * boundTask. The following parentheses make it a function call with the boundTask return type. The package task
74  * will then return a future of the correct type.
75  *
76  * Note: A std::packaged_task fulfills its future *during* the call to operator(). If the user of a
77  * std::packaged_task hands it off to another thread to execute, and then waits on the future, they will be able
78  * to retrieve the return value from the task and know that the task has executed, but they do not know exactly
79  * when the task object has been deleted. This distinction can be significant if the packaged task is holding
80  * onto resources that need to be freed (through a std::shared_ptr for example). If the user needs to wait for
81  * those resources to be freed they have no way of knowing how long to wait. The translated_task lambda below
82  * is a workaround for this limitation. It executes the packaged task, then disposes of it before passing the
83  * task's return value back to the future that the user is waiting on.
84  */
85  auto packaged_task = std::make_shared<ControlTask>(boundTask);
86 
87  // Create a promise/future that we will fulfill when we have cleaned up the task.
88  auto cleanupPromise = std::make_shared<std::promise<void>>();
89  auto cleanupFuture = cleanupPromise->get_future();
90 
91  // Remove the return type from the task by wrapping it in a lambda with no return value.
92  auto translated_task = [packaged_task, cleanupPromise]() mutable {
93  // Execute the task.
94  packaged_task->operator()();
95  // Note the future for the task's result.
96  auto taskFuture = packaged_task->get_future();
97  // Clean up the task.
98  packaged_task.reset();
99  // Forward the task's result to our cleanup promise/future.
100  forwardPromise(cleanupPromise, &taskFuture);
101  };
102 
103  // Release our local reference to packaged task so that the only remaining reference is inside the lambda.
104  packaged_task.reset();
105 
106  auto final_task = std::make_shared<ControlTask>(translated_task);
107  return final_task;
108  }
