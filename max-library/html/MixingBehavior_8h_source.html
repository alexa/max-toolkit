72 inline MixingBehavior getMixingBehavior(const std::string& input) {
73  MixingBehavior behavior = MixingBehavior::UNDEFINED;
74  if (mixingBehaviorToString(MixingBehavior::PRIMARY) == input) {
75  behavior = MixingBehavior::PRIMARY;
76  } else if (mixingBehaviorToString(MixingBehavior::MAY_DUCK) == input) {
77  behavior = MixingBehavior::MAY_DUCK;
78  } else if (mixingBehaviorToString(MixingBehavior::MUST_PAUSE) == input) {
79  behavior = MixingBehavior::MUST_PAUSE;
80  } else if (mixingBehaviorToString(MixingBehavior::MUST_STOP) == input) {
81  behavior = MixingBehavior::MUST_STOP;
82  }

84  return behavior;
85 }
